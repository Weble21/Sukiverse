port kill 명령어
# PID만 추출해서 바로 종료
$pid = (netstat -ano | findstr :8080 | Select-String 'LISTENING' | ForEach-Object { $_.ToString().Trim().Split()[-1] } | Select-Object -First 1)
taskkill /PID $pid /F

1단계-포트 사용 중인 PID 조회
netstat -ano | findstr :<포트번호>
# 예시
netstat -ano | findstr :8080

2단계-PID 프로세스 종료
taskkill /PID <PID번호> /F
# 예시
taskkill /PID 2492 /F

● 실행 방법

  코드 수정 후 재빌드할 때 (가장 자주 쓸 명령)
  docker-compose up --build -d

  처음 시작하거나 재시작할 때 (코드 변경 없을 때)
  docker-compose up -d

  중지할 때
  docker-compose down

  로그 볼 때
  docker logs sukiverse-app -f
